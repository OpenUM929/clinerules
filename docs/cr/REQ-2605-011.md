function getBorderColorFromClass(colorClass) {
    const match = colorClass.match(/^status-(CD\d+)/);
    if (!match) return '#4b5563';
    const cd = match[1];
    const info = settingsManager.getStatusInfoByCd(cd);
    return info?.txt_colr || '#4b5563';
}
